 * @author Özüm Eldoğan

var _       = require('lodash');
var sql     = require('../sql');
var pg      = require('pg');
var async   = require('async');


/**
 * Callback to handle database structure object.
 *
 * @callback exportCallback
 * @param {structure} Structure object of the database.
 */

/**
 * Exports a PostgreSQL schema as a detailed object.
 *
 * @param {String} host - Hostname of the database.
 * @param {String} database - Database name
 * @param {String} username - Username for connecting to db.
 * @param {String} password - Password to connecting to db.
 * @param {Object} options - Options
 * @param {String} [options.schema=public] - Schema of the database.
 * @param {Numeric} [options.port=5432] - Connection port of the database server.
 * @param {exportCallback} callback - Callback to handle database structure object.
 * @example
 * var pg-structure = require('pg-structure');
 * var util         = require('util');
 * pg-structure('localhost', 'db', 'user', 'password', { schema: 'public', port: 5432 }, function(result) {console.log(util.inspect(result, {depth: null}));} );
 * pg-structure('localhost', 'db', 'user', 'password', 'public', function(result) {console.log(JSON.stringify(result));} );
 */
module.exports = function(host, database, username, password, options, callback) {
    var structure = new Structure(host, database, username, password, options);
    structure.generate(callback);       // callback(result AS JSON)
}

var Structure = function(host, database, username, password, options) {
    options         = options || {};
    var port        = options.port || 5432;
    var schema      = options.schema || 'public';

    var result = {};
    var defaultTableObject = { fields: {}, fieldsOrdered: [], primaryKeys: [] }
    var Structure = this;

    var conString = 'postgres://' + username + ':' + password + '@' + host + ':' + port + '/' + database; // 'postgres://user:pass@host:port/db'

    /**
     * Generates the structure object and executes callback.
     * @param callback
     */
    this.generate = function(callback) {
        pg.connect(conString, function(err, client, done) {
            if (err) { return console.error('error fetching client from pool', err); }

            async.waterfall([
                client.query.bind(client, sql.field, [schema]),
                processFields,

                client.query.bind(client, sql.constraint, [schema]),
                processConstraints,

                client.query.bind(client, sql.index, [schema]),
                processIndexes,
                processThroughRelations,

            ], function (err) {
                if (err) { throw err; }
                done();
                callback(result);
            });
        });
    }

    /**
     * Processes result of the field SQL and populates result object.
     * @param {Object} dbResult - Result of the pg module query.
     * @param {Callback} next - Callback to execute after this function finishes its job.
     */
    function processFields(dbResult, next) {
        var field, table;

        for (var i = 0; i < dbResult.rows.length; i++) {
            field = dbResult.rows[i];
            if (!result[field.tableName]) {
                result[field.tableName] = _.cloneDeep( defaultTableObject );
                result[field.tableName].description = field.tableDescription;
                result[field.tableName].table = field.tableName;
            }
            table = result[field.tableName];
            if (field.default && field.default.toLowerCase().indexOf('nextval') !== -1 ) { field.isAutoIncrement = true }
            table.fields[field.name] = field;
            table.fieldsOrdered.push(field);
            delete field.tableSchema;
            delete field.tableName;
            delete field.tableDescription;
            delete field.position;
        }
        next();
    }

    /**
     * Processes result of the constraint SQL and populates result object.
     * @param {Object} dbResult - Result of the pg module query.
     * @param {Callback} next - Callback to execute after this function finishes its job.
     */
    function processConstraints(dbResult, next) {
        var constraint, table, field;

        for (var i = 0; i < dbResult.rows.length; i++) {
            constraint = dbResult.rows[i];
            table = result[constraint.tableName];
            field = table.fields[constraint.columnName]

            field.constraintName = constraint.constraintName;

            if (constraint.constraintType == 'PRIMARY KEY') {
                table.primaryKeys.push( field );
                field.isPrimaryKey = true;
            }
            else if (constraint.constraintType == 'FOREIGN KEY') {
                if (!table.foreignKeys ) { table.foreignKeys = [] }
                table.foreignKeys.push( field );
                field.isForeignKey = true;
                field.referencesTable   = constraint.referencesTable;
                field.referencesField   = constraint.referencesField;
                field.onUpdate          = constraint.onUpdate;
                field.onDelete          = constraint.onDelete;

                // Referenced Table hasMany
                if (!result[constraint.referencesTable] ) { result[constraint.referencesTable] = _.defaults( defaultTableObject, { hasMany: [] } ) }
                else if ( !result[constraint.referencesTable].hasMany ) { result[constraint.referencesTable].hasMany = [] }
                result[constraint.referencesTable].hasMany.push( { table: constraint.tableName, foreignKey: constraint.columnName } );
            }
            else if (constraint.constraintType == 'UNIQUE') {
                field.unique            = constraint.constraintName;
            }

        }
        next();
    }

    /**
     * Processes result of the index SQL and populates result object.
     * @param {Object} dbResult - Result of the pg module query.
     * @param {Callback} next - Callback to execute after this function finishes its job.
     */
    function processIndexes(dbResult, next) {
        var index;
        for (var i = 0; i < dbResult.rows.length; i++) {
            index = dbResult.rows[i];
            table = result[index.tableName];
            if (index.unique) {
                // String {name,surname} to Array ['name','surname'] sonra forEach
                index.columnNames.substr( 1, index.columnNames.length - 2).split(',').forEach(function(columnName) {
                    table.fields[columnName].unique = index.name;
                });
            }
        }
        next();
    }

    /**
     * Calculates through relations by examining result object.
     * @param {Callback} next - Callback to execute after this function finishes its job.
     */
    function processThroughRelations(next) {
        for ( var tableName in result ) {
            var table = result[tableName];
            if (! table.foreignKeys ) { continue }
            table.foreignKeys.forEach( function(firstFK ) {             // Tüm foreign key olan tablolardan diğer tüm
                var firstTable = result[firstFK.referencesTable];       // foreign key olan tablolara through yap.
                table.foreignKeys.forEach( function(secondFK )  {
                    var secondTable = result[secondFK.referencesTable];
                    if ( firstFK == secondFK ) { return }
                    if ( !firstTable.hasManyThrough ) { firstTable.hasManyThrough = [] }
                    firstTable.hasManyThrough.push( { table: secondFK.referencesTable, foreignKey: secondFK.name, through: tableName } )
                })
            })
        }
        next();
    }
};
